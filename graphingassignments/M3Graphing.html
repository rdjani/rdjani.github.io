fig,ax = plt.subplots()
x = df['Month-Year']
y = df['Total Construction']
ax.plot(x,y)

ax.spines['right'].set_visible(False)
ax.spines['top'].set_visible(False)

ax.set_xlabel('Dates (Month-Year)',fontsize='14',fontname='Times New Roman')
ax.set_ylabel('Total Construction',fontsize='14',fontname='Times New Roman')
ax.set_xticks(x[::20])
ax.set_xticklabels(x[::20])
ax.set_ylim(0,max(y)+10)

ax.axvline(x=80,linewidth=1,color='k',label='dip in construction start',linestyle='-')
ax.axvline(x=122,linewidth=1,color='k',label='dip in construction ends',linestyle='-')
ax.legend(loc=1)
ax.axvspan(80,122,ymin=0,ymax = 300,facecolor='k',alpha=0.1)
ax.text(102,40000,'Around the 2008 Reccession',fontsize=14,verticalalignment='center',horizontalalignment='center',fontname='Times New Roman')
